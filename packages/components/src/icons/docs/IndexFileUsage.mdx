<details>
    <summary>We recommend exporting your icons from <code>index.js</code> files</summary>

    ```jsx
    // assets/index.js

    import { createIcon } from "@sharegate/orbit-ui";
    import { ReactComponent as Azure } from "./icon-azure.svg";
    import { ReactComponent as Filter } from "./icon-filter.svg";

    export const AzureMajorIcon = createIcon(Azure);
    export const FilterMajorIcon = createIcon(Filter);
    ```

    ```jsx
    import { AzureMajorIcon, FilterMajorIcon } from "./assets";

    () => (
        <>
            <AzureMajorIcon />
            <FilterMajorIcon />
        </>
    );
    ```
